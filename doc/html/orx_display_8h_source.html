412 static orxINLINE orxRGBA orxColor_ToRGBA(const orxCOLOR *_pstColor)
413 {
414  orxRGBA stResult;
415  orxVECTOR vColor;
416  orxFLOAT fAlpha;
417 
418  /* Checks */
419  orxASSERT(_pstColor != orxNULL);
420 
421  /* Clamps RGB components */
422  orxVector_Clamp(&vColor, &(_pstColor->vRGB), &orxVECTOR_0, &orxVECTOR_WHITE);
423 
424  /* De-normalizes vector */
425  orxVector_Mulf(&vColor, &vColor, orxCOLOR_DENORMALIZER);
426 
427  /* Clamps alpha */

432

460 static orxCOLOR *orxFASTCALL orxColor_FromRGBToHSL(orxCOLOR *_pstDst, const orxCOLOR *_pstSrc)
461 {
462  orxCOLOR *pstResult = _pstDst;
463  orxFLOAT fMin, fMax, fDelta, fR, fG, fB;
464 
465  /* Checks */
466  orxASSERT(_pstDst != orxNULL);
467  orxASSERT(_pstSrc != orxNULL);
468 
469  /* Gets source red, blue and green components */
470  fR = _pstSrc->vRGB.fR;
471  fG = _pstSrc->vRGB.fG;
472  fB = _pstSrc->vRGB.fB;
473 
474  /* Gets min, max & delta values */
475  fMin = orxMIN(fR, orxMIN(fG, fB));
476  fMax = orxMAX(fR, orxMAX(fG, fB));
477  fDelta = fMax - fMin;
478 
479  /* Stores lightness */
480  pstResult->vHSL.fL = orx2F(0.5f) * (fMax + fMin);
481 
482  /* Gray? */
483  if(fDelta == orxFLOAT_0)
484  {
485  /* Gets hue & saturation */
486  pstResult->vHSL.fH = pstResult->vHSL.fS = orxFLOAT_0;
487  }
488  else
489  {
490  /* Updates saturation */
491  pstResult->vHSL.fS = (pstResult->vHSL.fL < orx2F(0.5f))
492  ? fDelta / (fMax + fMin)
493  : fDelta / (orx2F(2.0f) - fMax - fMin);
494 
495  /* Red tone? */
496  if(fR == fMax)
497  {
498  /* Updates hue */
499  pstResult->vHSL.fH = orx2F(1.0f / 6.0f) * (fG - fB) / fDelta;
500  }
501  /* Green tone? */
502  else if(fG == fMax)
503  {
504  /* Updates hue */
505  pstResult->vHSL.fH = orx2F(1.0f / 3.0f) + (orx2F(1.0f / 6.0f) * (fB - fR) / fDelta);
506  }
507  /* Blue tone */
508  else
509  {
510  /* Updates hue */
511  pstResult->vHSL.fH = orx2F(2.0f / 3.0f) + (orx2F(1.0f / 6.0f) * (fR - fG) / fDelta);
512  }
513 
514  /* Clamps hue */
515  if(pstResult->vHSL.fH < orxFLOAT_0)
516  {
517  pstResult->vHSL.fH += orxFLOAT_1;
518  }
519  else if(pstResult->vHSL.fH > orxFLOAT_1)
520  {
521  pstResult->vHSL.fH -= orxFLOAT_1;
522  }
523  }
524 
525  /* Updates alpha */
526  pstResult->fAlpha = _pstSrc->fAlpha;
527 
528  /* Done! */
529  return pstResult;
530 }

537 static orxCOLOR *orxFASTCALL orxColor_FromHSLToRGB(orxCOLOR *_pstDst, const orxCOLOR *_pstSrc)
538 {
539  orxCOLOR *pstResult = _pstDst;
540  orxFLOAT fH, fS, fL;
541 
542 #define orxCOLOR_GET_RGB_COMPONENT(RESULT, ALT, CHROMA, HUE) \
543 do \
544 { \
545  if(HUE < orx2F(1.0f / 6.0f)) \
546  { \
547  RESULT = ALT + (orx2F(6.0f) * HUE * (CHROMA - ALT)); \
548  } \
549  else if(HUE < orx2F(1.0f / 2.0f)) \
550  { \
551  RESULT = CHROMA; \
552  } \
553  else if(HUE < orx2F(2.0f / 3.0f)) \
554  { \
555  RESULT = ALT + (orx2F(6.0f) * (CHROMA - ALT) * (orx2F(2.0f / 3.0f) - HUE)); \
556  } \
557  else \
558  { \
559  RESULT = ALT; \
560  } \
561  if(RESULT < orxMATH_KF_EPSILON) \
562  { \
563  RESULT = orxFLOAT_0; \
564  } \
565  else if(RESULT > orxFLOAT_1 - orxMATH_KF_EPSILON) \
566  { \
567  RESULT = orxFLOAT_1; \
568  } \
569 } \
570 while(orxFALSE)
571 
572  /* Checks */
573  orxASSERT(_pstDst != orxNULL);
574  orxASSERT(_pstSrc != orxNULL);
575 
576  /* Gets source hue, saturation and lightness components */
577  fH = _pstSrc->vRGB.fH;
578  fS = _pstSrc->vRGB.fS;
579  fL = _pstSrc->vRGB.fL;
580 
581  /* Gray? */
582  if(fS == orxFLOAT_0)
583  {
584  /* Updates result */
585  orxVector_SetAll(&(pstResult->vRGB), fL);
586  }
587  else
588  {
589  orxFLOAT fChroma, fIntermediate;
590 
591  /* Gets chroma */
592  fChroma = (fL < orx2F(0.5f))
593  ? fL + (fL * fS)
594  : fL + fS - (fL * fS);
595 
596  /* Gets intermediate value */
597  fIntermediate = (orx2F(2.0f) * fL) - fChroma;
598 
599  /* Gets RGB components */
600  if(fH > orx2F(2.0f / 3.0f))
601  {
602  orxCOLOR_GET_RGB_COMPONENT(pstResult->vRGB.fR, fIntermediate, fChroma, (fH - orx2F(2.0f / 3.0f)));
603  }
604  else
605  {
606  orxCOLOR_GET_RGB_COMPONENT(pstResult->vRGB.fR, fIntermediate, fChroma, (fH + orx2F(1.0f / 3.0f)));
607  }
608  orxCOLOR_GET_RGB_COMPONENT(pstResult->vRGB.fG, fIntermediate, fChroma, fH);
609  if(fH < orx2F(1.0f / 3.0f))
610  {
611  orxCOLOR_GET_RGB_COMPONENT(pstResult->vRGB.fB, fIntermediate, fChroma, (fH + orx2F(2.0f / 3.0f)));
612  }
613  else
614  {
615  orxCOLOR_GET_RGB_COMPONENT(pstResult->vRGB.fB, fIntermediate, fChroma, (fH - orx2F(1.0f / 3.0f)));
616  }
617  }
618 
619  /* Updates alpha */
620  pstResult->fAlpha = _pstSrc->fAlpha;
621 
622  /* Done! */
623  return pstResult;
624 }

706 static orxCOLOR *orxFASTCALL orxColor_FromHSVToRGB(orxCOLOR *_pstDst, const orxCOLOR *_pstSrc)
707 {
708  orxCOLOR *pstResult = _pstDst;
709  orxFLOAT fH, fS, fV;
710 
711  /* Checks */
712  orxASSERT(_pstDst != orxNULL);
713  orxASSERT(_pstSrc != orxNULL);
714 
715  /* Gets source hue, saturation and value components */
716  fH = _pstSrc->vRGB.fH;
717  fS = _pstSrc->vRGB.fS;
718  fV = _pstSrc->vRGB.fV;
719 
720  /* Gray? */
721  if(fS == orxFLOAT_0)
722  {
723  /* Updates result */
724  orxVector_SetAll(&(pstResult->vRGB), fV);
725  }
726  else
727  {
728  orxFLOAT fFullHue, fSector, fIntermediate;
729 
730  /* Gets intermediate value */
731  fIntermediate = fV * fS;
732 
733  /* Gets full hue & sector */
734  fFullHue = orx2F(6.0f) * fH;
735  fSector = orxMath_Floor(fFullHue);
736 
737  /* Depending on sector */
738  switch(orxF2U(fSector))
739  {
740  default:
741  case 0:
742  {
743  /* Updates RGB components */
744  pstResult->vRGB.fR = fV;
745  pstResult->vRGB.fG = fV - (fIntermediate - (fIntermediate * (fFullHue - fSector)));
746  pstResult->vRGB.fB = fV - fIntermediate;
747 
748  break;
749  }
750 
751  case 1:
752  {
753  /* Updates RGB components */
754  pstResult->vRGB.fR = fV - (fIntermediate * (fFullHue - fSector));
755  pstResult->vRGB.fG = fV;
756  pstResult->vRGB.fB = fV - fIntermediate;
757 
758  break;
759  }
760 
761  case 2:
762  {
763  /* Updates RGB components */
764  pstResult->vRGB.fR = fV - fIntermediate;
765  pstResult->vRGB.fG = fV;
