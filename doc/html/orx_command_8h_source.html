103 do \
104 { \
105  orxCOMMAND_VAR_DEF stResult; \
106  orxCOMMAND_VAR_DEF astParamList[REQ_PARAM_NUMBER + OPT_PARAM_NUMBER + 1] = {__VA_ARGS__}; \
107  orxSTATUS eStatus; \
108  stResult.eType = RESULT_TYPE; \
109  stResult.zName = RESULT_NAME; \
110  eStatus = orxCommand_Register(#MODULE"."#COMMAND, orx##MODULE##_Command##COMMAND, REQ_PARAM_NUMBER, OPT_PARAM_NUMBER, astParamList, &stResult); \

112 } while(orxFALSE)
